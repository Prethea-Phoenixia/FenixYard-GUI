from msvcrt import getch
while True:
    keycode = ord(getch())
    if keycode == 27: #ESC
        break
    elif keycode == 13: #Enter
        select()
    elif keycode == 224: #Special keys (arrows, f keys, ins, del, etc.)
        keycode = ord(getch())
        if keycode == 80: #Down arrow
            moveDown()
        elif keycode == 72: #Up arrow
            moveUp()
